95inline void update_camera(Camera& cam, float dt, bool keys[1024], double mouseX, double mouseY) {
96 static double lastMouseX;
97 static double lastMouseY;
98 static bool firstMouse = true;
99
100 if (firstMouse) {
101 lastMouseX = mouseX;
102 lastMouseY = mouseY;
103 firstMouse = false;
104 }
105
106 float deltaX = float(mouseX - lastMouseX);
107 float deltaY = float(mouseY - lastMouseY);
108
109 lastMouseX = mouseX;
110 lastMouseY = mouseY;
111
112 cam.yaw += deltaX * cam.lookSpeed;
113 cam.pitch -= deltaY * cam.lookSpeed;
114
115 // Clamp pitch to prevent flipping
116 if (cam.pitch > 1.5708f) cam.pitch = 1.5708f;
117 if (cam.pitch < -1.5708f) cam.pitch = -1.5708f;
118
119 float cos_pitch = cosf(cam.pitch);
120 simd::float3 forward = simd::normalize(simd::float3{ sinf(cam.yaw) * cos_pitch, sinf(cam.pitch), -cosf(cam.yaw) * cos_pitch });
121 simd::float3 right = simd::normalize(simd::cross(forward, simd::float3{0, 1, 0}));
122
123 simd::float3 moveDir = {0, 0, 0};
124 if (keys['W']) moveDir += forward;
125 if (keys['S']) moveDir -= forward;
126 if (keys['A']) moveDir -= right;
127 if (keys['D']) moveDir += right;
128 if (keys[' ']) moveDir.y += 1.0f;
129 if (keys['C'] || keys['X']) moveDir.y -= 1.0f;
130
131 if (simd::length(moveDir) > 0.01f) {
132 cam.position += simd::normalize(moveDir) * cam.moveSpeed * dt;
133 }
134
135 // Clamp position to a bounding box
136 simd::float3 minBounds = {-20.0f, 0.0f, -20.0f};
137 simd::float3 maxBounds = {20.0f, 20.0f, 20.0f};
138 cam.position = simd::clamp(cam.position, minBounds, maxBounds);
139
140 cam.viewMatrix = matrix_look_at_right_hand(cam.position, cam.position + forward, simd::float3{0, 1, 0});
141}

145inline simd::float4x4 matrix_perspective_right_hand(float fovyRadians, float aspect, float nearZ, float farZ) {
146 float ys = 1.0f / tanf(fovyRadians * 0.5f);
147 float xs = ys / aspect;
148 float zs = farZ / (nearZ - farZ);
149 return simd::float4x4(
150 simd::float4{xs, 0.0f, 0.0f, 0.0f},
151 simd::float4{0.0f, ys, 0.0f, 0.0f},
152 simd::float4{0.0f, 0.0f, zs, -1.0f},
153 simd::float4{0.0f, 0.0f, zs * nearZ, 0.0f}
154 );
155}

157inline simd::float4x4 matrix_look_at_right_hand(simd::float3 eye, simd::float3 center, simd::float3 up) {
158 simd::float3 f = simd::normalize(center - eye);
159 simd::float3 s = simd::normalize(simd::cross(f, up));
160 simd::float3 u = simd::cross(s, f);
161
162 return simd::float4x4(
163 simd::float4{s.x, u.x, -f.x, 0.0f},
164 simd::float4{s.y, u.y, -f.y, 0.0f},
165 simd::float4{s.z, u.z, -f.z, 0.0f},
166 simd::float4{-simd::dot(s, eye), -simd::dot(u, eye), simd::dot(f, eye), 1.0f}
167 );
168}
